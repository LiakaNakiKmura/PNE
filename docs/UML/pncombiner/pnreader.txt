@startuml
abstract class Transaction{
    +excute()
}

abstract class PathAsk{
    +get_load_path()
}


class PathDialog{
    +get_load_path()
}

PathDialog --up|> PathAsk

abstract class Reader{
    +read()
}

class CSVIO{
    +read(message)
}

CSVIO --up|> Reader

class PNDataReader{
    +excute()
}

class PNPrmtrMng{
    +name getter
    +get_message()
}

class PNDataBase{
    <<singleton>>
    +set_noise()
}

PNDataReader --up|> Transaction
PNDataReader -right-> Reader
PNDataReader --> PNDataBase
PNDataReader --> PNPrmtrMng
CSVIO -right-> PathAsk


@enduml
